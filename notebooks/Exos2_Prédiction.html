# Mean Squared Error MSE en python

def mse(true_values, predicted_vals):
    values = 0
    for i in range(len(predicted_vals)):
        values += (predicted_vals[i] - true_values[i])**2
    print(values / len(predicted_vals))

# Mean Squared Error MSE en python avec numpy

def calcul_mse(true_values, predicted_vals):

    """ y_true : Valeurs r√©elles (observ√©es)
        y_pred : Valeurs pr√©dites par le mod√®le"""

    #convertir en tableaux numpy
    y_true=np.array(true_values)
    y_pred=np.array(predicted_vals)

    #calcul du MSE
    mse=np.mean((y_true-y_pred)**2)
    print("Mean Squared Error (MSE) =", mse)
    return mse
